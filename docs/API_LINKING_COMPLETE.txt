┌──────────────────────────────────────────────────────────────────┐
│ 🚀 HOW TO USE                                                    │
└──────────────────────────────────────────────────────────────────┘

1. Start Backend API:
   cd backend
   npm install
   npm run start:dev
   
   → Backend runs on http://process.env.NEXT_PUBLIC_API_URL
   → API Docs: http://process.env.NEXT_PUBLIC_API_URL/api/docs

2. Start Frontend:
   cd apps/web
   npm install
   npm run dev
   
   → Frontend runs on http://localhost:3000

3. Use in Frontend Components:

   // User Side Example
   import api from '@/lib/api';
   import { useBusinesses } from '@/hooks/useApi';
   
   function BusinessList() {
     const { businesses, isLoading } = useBusinesses({ city: 'Mumbai' });
     
     if (isLoading) return <Loading />;
     return <div>{businesses.map(b => <BusinessCard {...b} />)}</div>;
   }

   // Vendor Side Example
   import { useVendorStats, useMyBusinesses } from '@/hooks/useApi';
   
   function VendorDashboard() {
     const { stats } = useVendorStats();
     const { businesses } = useMyBusinesses();
     
     return (
       <div>
         <StatsCards stats={stats} />
         <BusinessList businesses={businesses} />
       </div>
     );
   }

   // Admin Side Example
   import { useAdminStats, useAllUsers } from '@/hooks/useApi';
   
   function AdminDashboard() {
     const { stats } = useAdminStats();
     const { users } = useAllUsers(1, 20);
     
     return (
       <div>
         <GlobalStats stats={stats} />
         <UserTable users={users} />
       </div>
     );
   }
